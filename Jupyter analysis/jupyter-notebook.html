times = list()
with open(PATH+"additionalMetrics.csv") as f:
    csv_reader = csv.reader(f, delimiter=',')
    line_count = 0
    for row in csv_reader:
        if line_count == 0:
            line_count += 1
        else:
            times.append(row[1])
            line_count += 1
times=list(map(int, times))
